    <script>
        const API_URL = "https://detect.roboflow.com/tri-blood-iw3al/18?api_key=jFEcRC9ZNlEy9YRvHfOJ";

        const DATA_MAP = {
            "SICKLE CELL": {
                label: "SICKLE CELL ANEMIA",
                features: "Crescent or sickle-shaped cells (drepanocytes) and polychromasia observed.",
                tips: "Ensure adequate hydration. Avoid extreme temperatures and overexertion. Monitor for pain crises — seek emergency care if breathing difficulty occurs.",
                todo: "Confirm with Hemoglobin Electrophoresis. Specialist referral required.",
                color: "text-orange-600", hex: "#ea580c", urgent: true
            },
            "THALASSEMIA": {
                label: "THALASSEMIA",
                features: "Microcytic, hypochromic cells with anisocytosis and poikilocytosis observed.",
                tips: "Avoid unsupervised iron supplementation until variant is confirmed. Report fatigue, pallor, or jaundice to a physician immediately.",
                todo: "Follow up with CBC and Iron Studies to confirm variant type.",
                color: "text-blue-600", hex: "#2563eb", urgent: true
            },
            "G6PD DEF.": {
                label: "G6PD DEFICIENCY",
                features: "Bite cells, blister cells, and ghost cells identified.",
                tips: "Avoid oxidative triggers: primaquine, dapsone, fava beans, and naphthalene. Watch for jaundice, dark urine, or sudden fatigue — signs of acute hemolysis.",
                todo: "Request G6PD enzyme assay; avoid oxidative medications without physician approval.",
                color: "text-green-600", hex: "#22c55e", urgent: true
            },
            "NORMAL RBC": {
                label: "NORMAL RBC",
                features: "Normocytic/Normochromic. Consistent 7-8µm size with 1/3 central pallor.",
                tips: "No morphological abnormalities detected. Maintain routine check-ups as recommended by your physician.",
                todo: "No immediate abnormalities detected.",
                color: "text-yellow-600", hex: "#facc15", urgent: false
            }
        };

        function resolveClass(rawClass) {
            const s = rawClass.toUpperCase().trim();
            if (s.includes("SICKLE"))                       return "SICKLE CELL";
            if (s.includes("THALASS"))                      return "THALASSEMIA";
            if (s.includes("G6PD") || s.includes("G 6PD")) return "G6PD DEF.";
            if (s.includes("NORMAL") || s.includes("RBC")) return "NORMAL RBC";
            return null;
        }

        function setStatus(text, classes) {
            const badge = document.getElementById("status-badge");
            badge.innerText = `AI STATUS: ${text}`;
            badge.className = `ai-status-badge ${classes}`;
        }

        async function analyze(imageSource) {
            setStatus("CONNECTING...", "bg-blue-100 text-blue-600 animate-pulse");
            const resBox = document.getElementById("result-box");

            try {
                const base64Data = imageSource.split(",")[1];
                const response = await fetch(API_URL, {
                    method: "POST",
                    body: base64Data,
                    headers: { "Content-Type": "application/x-www-form-urlencoded" }
                });

                if (!response.ok) throw new Error(`HTTP ${response.status}`);

                const data = await response.json();
                const canvas = document.getElementById("overlay");
                const ctx = canvas.getContext("2d");
                ctx.clearRect(0, 0, canvas.width, canvas.height);

                if (data.predictions && data.predictions.length > 0) {
                    setStatus("ONLINE", "bg-green-100 text-green-600");

                    data.predictions.forEach(p => {
                        const key = resolveClass(p.class);
                        const config = key ? DATA_MAP[key] : { hex: "#FF0000" };
                        ctx.strokeStyle = config.hex;
                        ctx.lineWidth = 4;
                        ctx.strokeRect(p.x - p.width / 2, p.y - p.height / 2, p.width, p.height);
                    });

                    const topKey = resolveClass(data.predictions[0].class);
                    const info = topKey ? DATA_MAP[topKey] : null;

                    if (!info) {
                        resBox.innerHTML = `
                            <h2 class="text-gray-600 font-black text-lg uppercase italic">Unclassified Morphology</h2>
                            <p class="text-xs font-bold text-gray-500 mt-2">Detected cells do not match known profiles. Refer to a specialist.</p>`;
                        return;
                    }

                    const urgentBanner = info.urgent
                        ? `<p class="text-[10px] font-black text-red-600 text-center mt-6 animate-pulse uppercase">⚠️ Pre-screening tool only — Seek Medical Professional Help Immediately</p>`
                        : `<p class="text-[10px] font-semibold text-gray-400 text-center mt-6 uppercase">Pre-screening result only. Confirm with a licensed physician.</p>`;

                    resBox.innerHTML = `
                        <div class="text-left w-full">
                            <h2 class="text-4xl font-black ${info.color} mb-4">${info.label}</h2>
                            <p class="text-[10px] font-black text-gray-400 uppercase">Morphology:</p>
                            <p class="text-sm font-bold text-gray-800 mb-4">${info.features}</p>
                            <div class="bg-amber-50 border border-amber-200 p-4 rounded-xl mb-3">
                                <p class="text-[10px] font-black text-amber-700 uppercase">Handling Tips:</p>
                                <p class="text-xs font-semibold text-gray-700 mt-1 leading-relaxed">${info.tips}</p>
                            </div>
                            <div class="bg-red-50 p-4 rounded-xl border border-red-100">
                                <p class="text-[10px] font-black text-red-600 uppercase">What to do:</p>
                                <p class="text-xs font-bold text-gray-700 mt-1">${info.todo}</p>
                            </div>
                            ${urgentBanner}
                        </div>`;

                } else {
                    setStatus("ONLINE", "bg-green-100 text-green-600");
                    resBox.innerHTML = `
                        <h2 class="text-red-600 font-black text-xl uppercase italic">Invalid Input</h2>
                        <p class="text-xs font-bold text-gray-500">No RBC markers identified.</p>`;
                }

            } catch (err) {
                console.error("Detection error:", err);
                setStatus("ERROR", "bg-red-100 text-red-600");
                resBox.innerHTML = `<p class="text-red-600 font-bold uppercase">Connection Failed. Check Internet.</p>`;
            }
        }

        document.getElementById("upload").onchange = (e) => {
            const reader = new FileReader();
            reader.onload = (f) => {
                const img = document.getElementById("uploaded-img");
                img.src = f.target.result;
                img.classList.remove("hidden");
                document.getElementById("placeholder-text").classList.add("hidden");
                document.getElementById("webcam").classList.add("hidden");

                const temp = new Image();
                temp.src = f.target.result;
                temp.onload = () => {
                    const canvas = document.getElementById("overlay");
                    canvas.width = temp.width;
                    canvas.height = temp.height;
                    analyze(f.target.result);
                };
            };
            reader.readAsDataURL(e.target.files[0]);
        };

        let stream = null;

        document.getElementById("webcam-btn").onclick = async () => {
            const v = document.getElementById("webcam");
            if (!stream) {
                stream = await navigator.mediaDevices.getUserMedia({ video: { facingMode: "environment" } });
                v.srcObject = stream;
                v.classList.remove("hidden");
                document.getElementById("snap-btn").classList.remove("hidden");
                document.getElementById("placeholder-text").classList.add("hidden");
            } else {
                stream.getTracks().forEach(t => t.stop());
                stream = null;
                v.classList.add("hidden");
                document.getElementById("snap-btn").classList.add("hidden");
            }
        };

        document.getElementById("snap-btn").onclick = () => {
            const v = document.getElementById("webcam");
            const c = document.createElement("canvas");
            c.width = v.videoWidth;
            c.height = v.videoHeight;
            c.getContext("2d").drawImage(v, 0, 0);
            document.getElementById("overlay").width = v.videoWidth;
            document.getElementById("overlay").height = v.videoHeight;
            analyze(c.toDataURL("image/jpeg", 0.8));
        };
    </script>
